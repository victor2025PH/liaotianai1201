## 🔍 立即檢查

執行以下命令檢查服務器狀態：

```bash
# 找到後端目錄
BACKEND_DIR=$(find ~ -type d -name "admin-backend" -exec test -f {}/app/main.py \; -print 2>/dev/null | head -1)

if [ -n "$BACKEND_DIR" ]; then
    echo "後端目錄: $BACKEND_DIR"
    cd "$BACKEND_DIR"
    
    # 檢查是否為 Git 倉庫
    if [ -d ".git" ]; then
        echo "✅ 是 Git 倉庫"
        echo "當前分支:"
        git branch
        echo ""
        echo "檢查代碼是否包含修復:"
        if grep -q "檢查數據庫（包含遠程服務器賬號）" app/api/group_ai/accounts.py; then
            echo "  ✅ 包含修復"
        else
            echo "  ❌ 不包含修復"
            echo ""
            echo "需要拉取最新代碼:"
            echo "  git pull origin main"
        fi
    else
        echo "⚠️ 不是 Git 倉庫，需要手動上傳文件"
    fi
else
    echo "❌ 未找到後端目錄"
fi
```
