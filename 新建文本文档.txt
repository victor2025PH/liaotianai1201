# 1. 重启所有 PM2 服务
pm2 restart all

# 2. 如果重启失败，先停止再启动
pm2 delete all
pm2 save --force

# 3. 启动后端
cd /home/ubuntu/telegram-ai-system/admin-backend
pm2 start start.sh --name backend

# 4. 启动前端
cd /home/ubuntu/telegram-ai-system/saas-demo
export PORT=3005
pm2 start npm --name saas-demo-frontend --update-env -- start

# 5. 启动其他三个服务
cd /home/ubuntu/telegram-ai-system
pm2 start npx --name tgmini-frontend -- serve -s tgmini20251220/dist -l 3001
pm2 start npx --name hongbao-frontend -- serve -s hbwy20251220/dist -l 3002
pm2 start npx --name aizkw-frontend -- serve -s aizkw20251219/dist -l 3003

# 6. 保存状态
pm2 save

# 7. 重启 Nginx
sudo systemctl restart nginx

# 8. 等待几秒后检查
sleep 5
pm2 list
